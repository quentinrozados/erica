147 /**
148 * @brief Die Bundesfinanzamtsnummer wird überprüft.
149 *
150 * Wird eine 13-stellige Steuernummer im
151 * ELSTER-Steuernummernformat angegeben, so wird nur die
152 * Bundesfinanzamtsnummer (= die ersten 4 Stellen der
153 * 13-stelligen Steuernummer) geprüft.
154 *
155 * Eine Prüfung der Steuernummer selbst findet nicht statt
156 * (hierfür EtkPruefeSteuernummer() verwenden).
157 *
158 * @param[in] steuernummer
159 * 13-stellige Steuernummer im ELSTER Steuernummernformat bzw.
160 * 4-stellige Bundesfinanzamtsnummer.
161 * @return
162 * - ::ERIC_OK
163 * - ::ERIC_GLOBAL_BUFANR_UNBEKANNT:
164 * Die Bundesfinanzamtsnummer ist unbekannt oder ungültig.
165 * - ::ERIC_GLOBAL_NULL_PARAMETER:
166 * Es wurde keine Bundesfinanzamtsnummer übergeben (Parameter ist NULL).
167 *
168 * @see
169 * - EtkPruefeSteuernummer()
170 * - Pruefung_der_Steuer_und_Steueridentifikatsnummer.pdf, siehe \linkExt{https://www.elster.de/elsterweb/entwickler/infoseite/schnittstellenbeschreibungen,Entwicklerbereich} bei \linkExt{https://www.elster.de/eportal/start,ELSTER}.
171 */
172 ETKAPI_DECL int STDCALL EtkPruefeBuFaNummer(const char* steuernummer);

174 /**
175 * @brief Die @c bic wird auf Gültigkeit überprüft.
176 *
177 * Die Prüfung erfolgt in zwei Schritten: @mAbsatz
178 * -# Formale Prüfung auf gültige Zeichen und richtige Länge
179 * -# Prüfung, ob das Länderkennzeichen für BIC gültig ist.
180 *
181 * @param[in] bic Zeiger auf eine NULL-terminierte Zeichenkette.
182 *
183 * @return
184 * - ::ERIC_OK
185 * - ::ERIC_GLOBAL_BIC_FORMALER_FEHLER: Ungültige Zeichen, falsche Länge.
186 * - ::ERIC_GLOBAL_BIC_LAENDERCODE_FEHLER
187 * - ::ERIC_GLOBAL_NULL_PARAMETER: Parameter @c bic ist NULL.
188 *
189 * @see
190 * - @typeDokumentation{ERiC-Entwicklerhandbuch.pdf}, Kap. "BIC ISO-Ländercodes"
191 * - @typeDokumentation{ERiC-Entwicklerhandbuch.pdf}, Kap. "BIC-Prüfung"
