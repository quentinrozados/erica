10/**
11 * @brief Handle auf eine Otto-Instanz
12 *
13 * Jede Funktion der Otto-API ist direkt oder indirekt an eine Otto-Instanz gebunden.
14 * Die Otto-Instanz enthält sämtliche veränderlichen Zustände von Otto.
15 *
16 * Es können mehrere Instanzen gleichzeitig existieren. Jede der
17 * Instanzen ist unabhängig von allen anderen. Verfügen mehrere Threads über
18 * jeweils ihre eigene Otto-Instanz, können sie diese Instanzen
19 * parallel verwenden.
20 *
21 * Eine Otto-Instanz soll nicht für jede Aufgabe neu erstellt und konfiguriert
22 * werden. Das Erstellen und Zerstören einer Otto-Instanz ist ressourcen- und zeitintensiv.
23 * Die Lebenszeit einer Otto-Instanz sollte beispielsweise eher der Lebenszeit
24 * eines Arbeiter-Threads in einem Pool entsprechen als der Verarbeitungsdauer
25 * einer einzelnen Aufgabe eines Arbeiter-Threads.
26 *
27 * Eine Otto-Instanz kann zwischen Threads ausgetauscht werden. Sie darf aber nicht
28 * in zwei Threads gleichzeitig verwendet werden.
29 *
30 * @see
31 * - OttoInstanzErzeugen()
32 * - OttoInstanzFreigeben()
33 */
34typedef struct OttoInstanz *OttoInstanzHandle;

90/**
91 * @brief Aufzählung der Log-Ebenen von Otto
92 *
93 * Die Log-Meldungen von Otto sind in Ebenen angeordnet:
94 * von der höchsten Ebene mit den wichtigsten Fehlermeldungen
95 * bis hin zu niedrigsten Ebene mit einfachen Meldungen, die
96 * nur bei der Suche nach Fehlerursachen interessant sind.
97 *
98 * Standardmäßig werden nur Meldungen der beiden höchsten Ebenen
99 * (Fehler und Warnungen) protokolliert.
100 *
101 * Sollen auch Meldungen der niedrigeren Ebenen protokolliert werden,
102 * so kann über die Funktion OttoInstanzErzeugen() ein Callback vom
103 * Typ ::OttoLogCallback registriert werden, in dem die Meldungen
104 * nach der Ebene gefiltert werden können.
105 *
106 * @see
107 * + OttoInstanzErzeugen()
108 */
109typedef enum {
110 /// Fehler, die zum Abbruch der gewünschten Aktion führen.
111 OTTOLOG_FEHLERMELDUNGEN = 4,
112
113 /// Hinweise auf Zustände, die zu Fehlern führen können.
114 OTTOLOG_WARNUNGEN = 3,
115
116 /// Grobe Informationen über den Programmablauf und Werte.
117 OTTOLOG_INFORMATIONEN = 2,
118
119 /// Feingranulare Informationen über den Programmablauf und Werte.
120 OTTOLOG_DEBUGMELDUNGEN = 1
121} OttoLogEbene;
